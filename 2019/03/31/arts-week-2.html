    def countPrimes(self, n: int) -> int:
        if n <= 2:
            return 0
        
        arr = [1] * n
        
        for i in range(2, int(n)):
            if arr[i] == 1:
                arr[2*i:n:i] = [0] * len(arr[2*i:n:i])
        
        return sum(arr[2:])
